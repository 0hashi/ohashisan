Apache


#!/usr/bin/python
#
# Script: apache
# This script is a wrapper for systemctl to manage the Apache2 service on kali00. I wrote this to help with
# multiple stops, starts, restarts of the Apache2 web server. Doesn't do much but it makes work a little
# faster for me.
#
# The PassPhrase for the SSL cert is: pe---------se (needed to start the service)
#

import sys
import subprocess

if len(sys.argv) == 1:
print ('You need more than', len(sys.argv), 'argument. Usage apache [status|start|stop|restart]')
sys.exit(1)

if str(sys.argv) != 'start' or str(sys.argv) != 'stop' or str(sys.argv) != 'status' or str(sys.argv) != 'restart':
print ('Usage: apache ')

if str(sys.argv[1]) == 'start':
print ('Gonna start apache', str(sys.argv[1]))
cmd_str = "systemctl enable apache2;systemctl start apache2"
subprocess.run(cmd_str, shell=True)

elif str(sys.argv[1]) == 'stop':
print ('Gonna stop apache')
cmd_str = "systemctl stop apache2;systemctl disable apache2"
subprocess.run(cmd_str, shell=True)

elif str(sys.argv[1]) == 'restart':
print ('Gonna restart apache')
cmd_str = "systemctl restart apache2"
subprocess.run(cmd_str, shell=True)

elif str(sys.argv[1]) == 'status':
print ('Checking apache status')
cmd_str = "systemctl status apache2"
subprocess.run(cmd_str, shell=True)
