Bulk Set-ADUser attributes for all users in defined OU



# Bulk Set-ADUser attribute for all users in defined OU
# Import the Active Directory module

Import-Module ActiveDirectory

# Define the OU path where you want to list and update users
$ouPath = "OU=Users,OU=NAM,OU=Enterprise,OU=www.reddirt.lab,OU=Active Roles,OU=Solutions,DC=reddirt,DC=lab"

# Define the value you want to set for the Office attribute
$newOfficeValue = "NAM"

# Get a list of users in the specified OU
$users = Get-ADUser -Filter * -SearchBase $ouPath

# Loop through each user and set the Office attribute
foreach ($user in $users) {
# Display the user's current attributes (optional)
Write-Host "Current Office attribute for $($user.Name): $($user.Office)"

# Set the new Office attribute value
Set-ADUser -Identity $user -Office $newOfficeValue

# Display a confirmation message (optional)
Write-Host "Updated Office attribute for $($user.Name) to $newOfficeValue"
}

# Optional: Display a completion message
Write-Host "Script completed."
